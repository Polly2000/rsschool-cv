function twoMaxNums(arr) {
  let max1 = Math.max(arr[0], arr[1]);
  let max2 = Math.max(arr[0], arr[1]);

  for (let i = 2; i < arr.length; i++) {
    if (arr[i] > max1) {
      max1 = arr[i];
    } else if (arr[i] > max2) {
      max2 = arr[i];
    }
  }

  return { max1, max2 };
}
